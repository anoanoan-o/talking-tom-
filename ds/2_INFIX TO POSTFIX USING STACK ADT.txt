#include <stdio.h>
#include <ctype.h>
#define MAX 50

char st[MAX];
int top = -1;

void push(char c){ st[++top] = c; }
char pop(){ return st[top--]; }
int prec(char c){
    if(c=='^') return 3;
    if(c=='*'||c=='/') return 2;
    if(c=='+'||c=='-') return 1;
    return 0;
}

int main(){
    char infix[MAX], postfix[MAX], ch;
    int i=0, k=0;
    printf("Enter Infix: ");
    scanf("%s", infix);

    for(i=0; infix[i]!='\0'; i++){
        ch = infix[i];
        if(isalnum(ch)) postfix[k++] = ch;
        else if(ch=='(') push(ch);
        else if(ch==')'){
            while(st[top]!='(') postfix[k++] = pop();
            pop();
        } else {
            while(top!=-1 && prec(st[top])>=prec(ch))
                postfix[k++] = pop();
            push(ch);
        }
    }
    while(top!=-1) postfix[k++] = pop();
    postfix[k]='\0';

    printf("Postfix: %s\n", postfix);
    return 0;
}
