 #include <stdio.h>
#include <ctype.h>
#define MAX 50

int st[MAX], top = -1;

void push(int v){ st[++top] = v; }
int pop(){ return st[top--]; }

int main(){
    char exp[50];
    int i, a, b;
    printf("Enter Postfix Expression: ");
    scanf("%s", exp);

    for(i=0; exp[i]!='\0'; i++){
        if(isdigit(exp[i])) 
            push(exp[i]-'0');          // convert char to int
        else{
            b = pop();
            a = pop();
            switch(exp[i]){
                case '+': push(a+b); break;
                case '-': push(a-b); break;
                case '*': push(a*b); break;
                case '/': push(a/b); break;
            }
        }
    }
    printf("Result = %d\n", st[top]);
    return 0;
}
