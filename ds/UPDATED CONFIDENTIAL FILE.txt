1.IMPLEMENT STACK USING ARRAYS
#include <stdio.h>
#define MAX 3
int st[MAX], top=-1;

void push(int v){ 
    if(top==MAX-1) printf("Overflow\n"); 
    else st[++top]=v; 
}
int pop(){ 
    if(top==-1){ printf("Underflow\n"); return -1; } 
    return st[top--]; 
}
void display(){ 
    if(top==-1) printf("Empty\n"); 
    else for(int i=top;i>=0;i--) printf("%d\n",st[i]); 
}
int peek(){ 
    return (top==-1)? (printf("Empty\n"),-1):st[top]; 
}

int main(){
    int ch,v;
    do{
        printf("\n1.PUSH 2.POP 3.PEEK 4.DISPLAY 5.EXIT\nChoice: ");
        scanf("%d",&ch);
        switch(ch){
            case 1: printf("Enter: "); scanf("%d",&v); push(v); break;
            case 2: v=pop(); if(v!=-1) printf("Popped %d\n",v); break;
            case 3: v=peek(); if(v!=-1) printf("Top %d\n",v); break;
            case 4: display(); break;
        }
    }while(ch!=5);
}



2.INFIX TO POSTFIX USING STACK ADT
#include <stdio.h>
#include <ctype.h>
#define MAX 50

char st[MAX];
int top = -1;

void push(char c){ st[++top] = c; }
char pop(){ return st[top--]; }
int prec(char c){
    if(c=='^') return 3;
    if(c=='*'||c=='/') return 2;
    if(c=='+'||c=='-') return 1;
    return 0;
}

int main(){
    char infix[MAX], postfix[MAX], ch;
    int i=0, k=0;
    printf("Enter Infix: ");
    scanf("%s", infix);

    for(i=0; infix[i]!='\0'; i++){
        ch = infix[i];
        if(isalnum(ch)) postfix[k++] = ch;
        else if(ch=='(') push(ch);
        else if(ch==')'){
            while(st[top]!='(') postfix[k++] = pop();
            pop();
        } else {
            while(top!=-1 && prec(st[top])>=prec(ch))
                postfix[k++] = pop();
            push(ch);
        }
    }
    while(top!=-1) postfix[k++] = pop();
    postfix[k]='\0';

    printf("Postfix: %s\n", postfix);
    return 0;
}



3. POSTFIX EXPRESSION USING STACK ADT
 #include <stdio.h>
#include <ctype.h>
#define MAX 50

int st[MAX], top = -1;

void push(int v){ st[++top] = v; }
int pop(){ return st[top--]; }

int main(){
    char exp[50];
    int i, a, b;
    printf("Enter Postfix Expression: ");
    scanf("%s", exp);

    for(i=0; exp[i]!='\0'; i++){
        if(isdigit(exp[i])) 
            push(exp[i]-'0');          // convert char to int
        else{
            b = pop();
            a = pop();
            switch(exp[i]){
                case '+': push(a+b); break;
                case '-': push(a-b); break;
                case '*': push(a*b); break;
                case '/': push(a/b); break;
            }
        }
    }
    printf("Result = %d\n", st[top]);
    return 0;
}



4. LINEAR QUEUE ADT USING ARRAY
#include <stdio.h>
#define MAX 5

int q[MAX], front = -1, rear = -1;

void enqueue(int val) {
    if (rear == MAX - 1)
        printf("Queue Overflow\n");
    else {
        if (front == -1) front = 0;
        q[++rear] = val;
    }
}

void dequeue() {
    if (front == -1 || front > rear)
        printf("Queue Underflow\n");
    else
        printf("Deleted: %d\n", q[front++]);
}

void display() {
    if (front == -1 || front > rear)
        printf("Queue Empty\n");
    else {
        for (int i = front; i <= rear; i++)
            printf("%d ", q[i]);
        printf("\n");
    }
}

int main() {
    int ch, val;
    do {
        printf("\n1.Enqueue 2.Dequeue 3.Display 4.Exit\nEnter choice: ");
        scanf("%d", &ch);
        switch (ch) {
            case 1: printf("Enter value: "); scanf("%d", &val); enqueue(val); break;
            case 2: dequeue(); break;
            case 3: display(); break;
        }
    } while (ch != 4);
    return 0;
}



5.CIRCULAR QUEUE ADT USING ARRAY
#include <stdio.h>
#define MAX 5

int q[MAX], front = -1, rear = -1;

void enqueue(int val) {
    if ((rear + 1) % MAX == front)
        printf("Queue Overflow\n");
    else {
        if (front == -1) front = 0;
        rear = (rear + 1) % MAX;
        q[rear] = val;
    }
}

void dequeue() {
    if (front == -1)
        printf("Queue Underflow\n");
    else {
        printf("Deleted: %d\n", q[front]);
        if (front == rear)
            front = rear = -1;
        else
            front = (front + 1) % MAX;
    }
}

void display() {
    if (front == -1)
        printf("Queue Empty\n");
    else {
        int i = front;
        while (1) {
            printf("%d ", q[i]);
            if (i == rear) break;
            i = (i + 1) % MAX;
        }
        printf("\n");
    }
}

int main() {
    int ch, val;

        printf("\n1.Enqueue 2.Dequeue 3.Display 4.Exit\nEnter choice: ");
        scanf("%d", &ch);

            case 1: printf("Enter value: "); scanf("%d", &val); enqueue(val); break;
            case 2: dequeue(); break;

    return 0;
}


6.CIRCULAR LINKED LIST ADT
#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next;
};
struct node *last = NULL;

void insertEnd(int val) {
    struct node *newNode = malloc(sizeof(struct node));
    newNode->data = val;
    if (last == NULL) {
        last = newNode;
        newNode->next = last;
    } else {
        newNode->next = last->next;
        last->next = newNode;
        last = newNode;
    }
}

void deleteFront() {
    if (last == NULL)
        printf("List Empty\n");
    else {
        struct node *temp = last->next;
        if (temp == last)
            last = NULL;
        else
            last->next = temp->next;
        printf("Deleted: %d\n", temp->data);
        free(temp);
    }
}

void display() {
    if (last == NULL)
        printf("List Empty\n");
    else {
        struct node *temp = last->next;
        do {
            printf("%d ", temp->data);
            temp = temp->next;
        } while (temp != last->next);
        printf("\n");
    }
}

int main() {
    int ch, val;
    do {
        printf("\n1.Insert 2.Delete 3.Display 4.Exit\nEnter choice: ");
        scanf("%d", &ch);
        switch (ch) {
            case 1: printf("Enter value: "); scanf("%d", &val); insertEnd(val); break;
            case 2: deleteFront(); break;
            case 3: display(); break;
        }
    } while (ch != 4);
    return 0;
}


7.STACK/LINEAR QUEUE ADT USING LINKED LIST
#include <stdio.h>
#include <stdlib.h>

struct Node { int data; struct Node* next; };
struct Node *head=NULL, *top=NULL, *front=NULL, *rear=NULL;

void insertLL(int x){ struct Node* n=malloc(sizeof(struct Node)); n->data=x; n->next=head; head=n; }
void deleteLL(){ if(!head){ printf("List Empty\n"); return; } struct Node* t=head; printf("Deleted: %d\n",t->data); head=head->next; free(t); }
void displayLL(){ struct Node* t=head; if(!t){ printf("List Empty\n"); return; } printf("LL: "); while(t){ printf("%d ",t->data); t=t->next; } printf("\n"); }

void push(int x){ struct Node* n=malloc(sizeof(struct Node)); n->data=x; n->next=top; top=n; }
void pop(){ if(!top){ printf("Stack Empty\n"); return; } struct Node* t=top; printf("Popped: %d\n",t->data); top=top->next; free(t); }
void displayStack(){ struct Node* t=top; if(!t){ printf("Stack Empty\n"); return; } printf("Stack: "); while(t){ printf("%d ",t->data); t=t->next; } printf("\n"); }

void enqueue(int x){ struct Node* n=malloc(sizeof(struct Node)); n->data=x; n->next=NULL; if(!rear) front=rear=n; else { rear->next=n; rear=n; } }
void dequeue(){ if(!front){ printf("Queue Empty\n"); return; } struct Node* t=front; printf("Dequeued: %d\n",t->data); front=front->next; if(!front) rear=NULL; free(t); }
void displayQueue(){ struct Node* t=front; if(!t){ printf("Queue Empty\n"); return; } printf("Queue: "); while(t){ printf("%d ",t->data); t=t->next; } printf("\n"); }

int main(){
    int ch,val;
    do{
        printf("\n1.Insert LL 2.Delete LL 3.Disp LL\n4.Push 5.Pop 6.Disp Stack\n7.Enqueue 8.Dequeue 9.Disp Queue 10.Exit\nEnter: ");
        scanf("%d",&ch);
        switch(ch){
            case 1: scanf("%d",&val); insertLL(val); break;
            case 2: deleteLL(); break;
            case 3: displayLL(); break;
            case 4: scanf("%d",&val); push(val); break;
            case 5: pop(); break;

8. Binary Search Tree ADT using Linked List
#include <stdio.h>
#include <stdlib.h>
struct Node {
int key;
struct Node *left, *right;
};
struct Node* newNode(int item) {
struct Node* temp = (struct Node*)malloc(sizeof(struct Node));
temp->key = item;
temp->left = temp->right = NULL;
return temp;
}
struct Node* insert(struct Node* node, int key) {
if (node == NULL)
return newNode(key);
if (key < node->key)
node->left = insert(node->left, key);
else if (key > node->key)
node->right = insert(node->right, key);
return node;
}
void inorder(struct Node* root) {
if (root != NULL) {
inorder(root->left);
printf("%d ", root->key);
inorder(root->right);
}
}
void preorder(struct Node* root) {
if (root != NULL) {
printf("%d ", root->key);
preorder(root->left);
preorder(root->right);
}
}
void postorder(struct Node* root) {
if (root != NULL) {
postorder(root->left);
postorder(root->right);
printf("%d ", root->key);
}
}
int main() {
struct Node* root = NULL;
int n, val;
printf("Enter number of elements to insert: ");
scanf("%d", &n);
printf("Enter %d elements:\n", n);
for (int i = 0; i < n; i++) {
scanf("%d", &val);

}
printf("Inorder: ");
inorder(root);
printf("\n");
printf("Preorder: ");
preorder(root);
printf("\n");
printf("Postorder: ");
postorder(root);
printf("\n");
return 0;
}

9. Graph Traversal techniques: a) Depth First Search b) Breadth First Search
#include <stdio.h>
#include <stdlib.h>
#define MAX 100
int adj[MAX][MAX];
int visited[MAX];
int n; // number of vertices
void DFS(int v) {
visited[v] = 1;
printf("%d ", v);
for (int i = 0; i < n; i++) {
if (adj[v][i] && !visited[i])
DFS(i);
}
}
void BFS(int start) {
int queue[MAX], front = 0, rear = 0;
for (int i = 0; i < n; i++) visited[i] = 0;
queue[rear++] = start;
visited[start] = 1;
while (front < rear) {
int v = queue[front++];
printf("%d ", v);
for (int i = 0; i < n; i++) {
if (adj[v][i] && !visited[i]) {
queue[rear++] = i;
visited[i] = 1;
}
}
}
}
int main() {
int e, v1, v2, start;
printf("Enter number of vertices: ");
scanf("%d", &n);
printf("Enter number of edges: ");
scanf("%d", &e);
// initialize adjacency matrix
for (int i = 0; i < n; i++)
for (int j = 0; j < n; j++)
adj[i][j] = 0;
printf("Enter %d edges (as pairs u v):\n", e);
for (int i = 0; i < e; i++) {
scanf("%d %d", &v1, &v2);
adj[v1][v2] = 1;
adj[v2][v1] = 1; // For undirected graph
}
printf("Enter starting vertex for traversal: ");
scanf("%d", &start);
printf("DFS traversal: ");
for (int i = 0; i < n; i++) visited[i] = 0;
DFS(start);
printf("\n");
printf("BFS traversal: ");
BFS(start);
printf("\n");
return 0;
}

10. Applications of Binary Search Technique
#include <stdio.h>
int binarySearch(int arr[], int n, int target) {
int left = 0, right = n - 1;
while (left <= right) {
int mid = left + (right - left) / 2;
if (arr[mid] == target)
return mid; // Target found
else if (arr[mid] < target)
left = mid + 1;
else
right = mid - 1;
}
return -1;
}
int main() {
int n, i, target;
printf("Enter the number of product IDs in the inventory: ");
scanf("%d", &n);
int productIDs[n];
printf("Enter %d sorted product IDs:\n", n);
for (i = 0; i < n; i++) {
scanf("%d", &productIDs[i]);
}
printf("Enter the product ID to search: ");
scanf("%d", &target);
int result = binarySearch(productIDs, n, target);
if (result != -1)
printf("Product ID %d is available at index %d.\n", target, result);
else
printf("Product ID %d is not available in the inventory.\n", target);
return 0;
}
