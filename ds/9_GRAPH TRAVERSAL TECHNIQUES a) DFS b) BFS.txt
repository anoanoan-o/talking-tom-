#include <stdio.h>
#include <stdlib.h>
#define MAX 100
int adj[MAX][MAX];
int visited[MAX];
int n; // number of vertices
void DFS(int v) {
visited[v] = 1;
printf("%d ", v);
for (int i = 0; i < n; i++) {
if (adj[v][i] && !visited[i])
DFS(i);
}
}
void BFS(int start) {
int queue[MAX], front = 0, rear = 0;
for (int i = 0; i < n; i++) visited[i] = 0;
queue[rear++] = start;
visited[start] = 1;
while (front < rear) {
int v = queue[front++];
printf("%d ", v);
for (int i = 0; i < n; i++) {
if (adj[v][i] && !visited[i]) {
queue[rear++] = i;
visited[i] = 1;
}
}
}
}
int main() {
int e, v1, v2, start;
printf("Enter number of vertices: ");
scanf("%d", &n);
printf("Enter number of edges: ");
scanf("%d", &e);
// initialize adjacency matrix
for (int i = 0; i < n; i++)
for (int j = 0; j < n; j++)
adj[i][j] = 0;
printf("Enter %d edges (as pairs u v):\n", e);
for (int i = 0; i < e; i++) {
scanf("%d %d", &v1, &v2);
adj[v1][v2] = 1;
adj[v2][v1] = 1; // For undirected graph
}
printf("Enter starting vertex for traversal: ");
scanf("%d", &start);
printf("DFS traversal: ");
for (int i = 0; i < n; i++) visited[i] = 0;
DFS(start);
printf("\n");
printf("BFS traversal: ");
BFS(start);
printf("\n");
return 0;
}
