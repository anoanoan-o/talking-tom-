#Write a program in python to explore threading using functions.
import threading
def print_numbers():
    for i in range(5):
        print(f"Number: {i}")
def print_letters():
    for letter in ['A', 'B', 'C', 'D', 'E']:
        print(f"\n Letter: {letter}")
t1 = threading.Thread(target=print_numbers)
t2 = threading.Thread(target=print_letters)
t1.start()
t2.start()
t1.join()
t2.join()
print("Main thread finished.")

#Write a program to implement threading with random period of time
import threading
import time
import random
def worker(name):
    sleep_time = random.randint(1, 5)
    print(f"{name} will sleep for {sleep_time} seconds.")
    time.sleep(sleep_time)
    print(f"{name} finished after {sleep_time} seconds.")
threads = []
for i in range(3):
    t = threading.Thread(target=worker, args=(f"Thread-{i+1}",))
    threads.append(t)
    t.start()
for t in threads:
    t.join()
print("All threads completed.")
